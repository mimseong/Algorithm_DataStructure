    for (int i = 1; i < v.size(); i++)
    {
        int temp = v[i];
        int prev = i - 1;
        
        while ((prev >= 0) && (v[prev] > temp))
        {
            v[prev + 1] = v[prev];
            prev--;
        }
        v[prev + 1] = temp;
    }
